---
title: Get profile
description: Retrieve the authenticated user's profile information
---

## Endpoint

```
GET /api/users/profile
```

## Authentication

This endpoint requires authentication. Include a valid JWT token in the Authorization header:

```
Authorization: Bearer <token>
```

## Response

### Success response

Returns the user's profile data, including store information if the user is a store employee.

<ResponseField name="success" type="boolean" required>
  Indicates if the request was successful
</ResponseField>

<ResponseField name="data" type="object" required>
  User profile data
  
  <ResponseField name="id" type="string" required>
    User's unique identifier
  </ResponseField>
  
  <ResponseField name="name" type="string" required>
    User's full name
  </ResponseField>
  
  <ResponseField name="email" type="string" required>
    User's email address
  </ResponseField>
  
  <ResponseField name="role" type="string" required>
    User's role. One of: `student`, `faculty`, or `store_employee`
  </ResponseField>
  
  <ResponseField name="register_number" type="string">
    Student or faculty registration number (null if not applicable)
  </ResponseField>
  
  <ResponseField name="employee_id" type="string">
    Employee ID for store employees (null if not applicable)
  </ResponseField>
  
  <ResponseField name="phone_number" type="string">
    User's phone number (null if not provided)
  </ResponseField>
  
  <ResponseField name="is_email_verified" type="boolean" required>
    Whether the user's email has been verified
  </ResponseField>
  
  <ResponseField name="no_show_count" type="number" required>
    Number of times the user failed to pick up orders
  </ResponseField>
  
  <ResponseField name="trust_tier" type="string" required>
    User's trust level based on order history. One of: `good`, `watch`, or `restricted`
  </ResponseField>
  
  <ResponseField name="ordering_restricted_until" type="string">
    ISO 8601 date when ordering restrictions will be lifted (null if not restricted)
  </ResponseField>
  
  <ResponseField name="last_no_show_at" type="string">
    ISO 8601 date of the last no-show incident (null if none)
  </ResponseField>
  
  <ResponseField name="created_at" type="string" required>
    ISO 8601 date when the account was created
  </ResponseField>
  
  <ResponseField name="updated_at" type="string" required>
    ISO 8601 date when the profile was last updated
  </ResponseField>
  
  <ResponseField name="store" type="object">
    Store information (only present if user role is `store_employee` and owns a store)
  </ResponseField>
</ResponseField>

### Error responses

<ResponseField name="success" type="boolean" required>
  Always `false` for errors
</ResponseField>

<ResponseField name="message" type="string" required>
  Error description
</ResponseField>

## Examples

<CodeGroup>

```bash curl
curl -X GET https://api.campusbite.com/api/users/profile \
  -H "Authorization: Bearer YOUR_JWT_TOKEN"
```

```javascript JavaScript
const response = await fetch('https://api.campusbite.com/api/users/profile', {
  method: 'GET',
  headers: {
    'Authorization': 'Bearer YOUR_JWT_TOKEN'
  }
});

const data = await response.json();
console.log(data);
```

```python Python
import requests

response = requests.get(
    'https://api.campusbite.com/api/users/profile',
    headers={'Authorization': 'Bearer YOUR_JWT_TOKEN'}
)

data = response.json()
print(data)
```

</CodeGroup>
